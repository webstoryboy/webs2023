Q : 100이하의 자연수 A, B, C를 입력받아 가장 작은 수를 출력하시오.

function solution(a, b, c){
    let answer;
    
    if( a < b ) {
        answer = a;
    } else {
        answer = b;
    }
    if( c < answer ) {
        answer = c;
    }
    return answer;
}
console.log(solution(6, 5, 5))
