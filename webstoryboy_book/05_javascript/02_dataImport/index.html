자바스크립트 : 데이터 출력하기

난이도 쉬움

소개

안녕하세요!. 웹스토리보이입니다. 자바스크립트를 통해 데이터 저장하는 방법을 배웠으면, 이제는 출력하는 방법 즉 불러오는 방법에 대해서 배우겠습니다.

01. 변수를 통해 데이터를 출력합니다.

{
    let x = 100, y = 200, z = "javascript";

    document.write(x);
    document.write(y);
    document.write(z);

    //100 200 javascript
}

02. 상수를 통해 데이터를 출력합니다.

{
    const x = 100, y = 200, z = "javascript";  

    document.write(x);
    document.write(y);
    document.write(z);

    //100 200 javascript
}

03. 배열을 통해 데이터를 출력합니다.

{
    const arr = [100, 200, "javascript"];

    document.write(arr[0]);
    document.write(arr[1]);
    document.write(arr[2]);

    //100 200 javascript
}

04. 2차 배열을 통해 데이터를 출력합니다.

{
    const arr = [100, 200, ["javascript", "jquery"]];

    document.write(arr[0]);
    document.write(arr[1]);
    document.write(arr[2]);
    document.write(arr[2][0]);
    document.write(arr[2][1]);

    //100
    //200
    //javascript,jquery
    //javascript
    //jquery
}

05. 배열의 데이터 갯수를 출력합니다.

{
    const arr = [100, 200, "javascript"];

    document.write(arr.length);

    //3
}

06. 배열의 데이터를 for문을 통해 불러옵니다.

{
    const arr = [100, 200, 300, 400, 500];

    // document.write(arr[0])
    // document.write(arr[1])
    // document.write(arr[2])
    // document.write(arr[3])
    // document.write(arr[4])

    //for(초기값, 조건식, 증감식)
    for( let i=0; i<arr.length; i++ ){
        document.write(arr[i]);
    }

    //100, 200, 300, 400, 500
}

07. 배열의 데이터를 forEach를 통해 불러옵니다.

{
    const num = [100, 200, 300, 400, 500];

    document.write(num[0])
    document.write(num[1])
    document.write(num[2])
    document.write(num[3])
    document.write(num[4])

    //for문
    for( let i=0; i<num.length; i++ ){
        document.write(num[i]);
    }

    //forEach()
    num.forEach(function(el){
        document.write(el);
    });

    //forEach(element, index, array)
    num.forEach(function(element, index, array){
        document.write(element);
        document.write(index);
        document.write(array);
    });
}

08. 배열의 데이터를 for of를 통해 불러옵니다.

{
    const num = [100, 200, 300, 400, 500];

    for( let i of num ){
        document.write(i);
    }
}

09. 배열의 데이터를 for in을 통해 불러옵니다.

{
    const num = [100, 200, 300, 400, 500];

    for( let i in num ){
        document.write(i);
    }

    for( let i in num ){
        document.write(num[i]);
    }
}

10. 배열의 데이터를 map()를 통해 불러옵니다.

{
    const num = [100, 200, 300, 400, 500];
    
    //map()
    num.map(function(el){
        document.write(el);
    });

    //map(element, index, array)
    num.map(function(element, index, array){
        document.write(element);
        document.write(index);
        document.write(array);
    });
}

11. 배열의 데이터를 펼침 연산자를 통해 불러옵니다.

{
    const num = [100, 200, 300, 400, 500];
    
    document.write(num);
    document.write(num[0], num[1], num[2], num[3], num[4]);
    document.write(...num);
}

12. 배열의 데이터를 배열구조 할당을 통해 불러옵니다.

{
    let a, b, c;

    [a, b, c] = [100, 200, "javascript"];

    document.write(a);
    document.write(b);
    document.write(c);
}

13. 객체 데이터를 불러옵니다.

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    }
    
    document.write(obj.a);
    document.write(obj.b);
    document.write(obj.c);
    document.write(obj['a']);
    document.write(obj['b']);
    document.write(obj['c']);
}

14. 객체 데이터를 불러옵니다.

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    }

    document.write(Object.keys(obj));
    document.write(Object.values(obj));
    document.write(Object.entries(obj));
}

15. 객체 데이터를 변수에 저장하여 불러옵니다.

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    }
    const name1 = obj.a;
    const name2 = obj.b;
    const name3 = obj.c;

    document.write(name1);
    document.write(name2);
    document.write(name3);
}

16. 객체 데이터를 fon in문을 통해 불러옵니다.

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    }
    
    for( let key in obj ){
        document.write(obj[key]);
    }
}

17. 객체 데이터를 map()을 통해 불러옵니다.

{
    const obj = [
        {a:100, b:200, c:"javascript1"}
    ];

    obj.map((el) => {
        document.write(el.a);
        document.write(el.b);
        document.write(el.c);
    });
}

18. 객체 데이터를 hasOwnProperty()을 통해 불러옵니다.

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    }

    document.write(obj.hasOwnProperty('a'));
    document.write(obj.hasOwnProperty('b'));
    document.write(obj.hasOwnProperty('c'));
    document.write(obj.hasOwnProperty('d'));
    document.write('a' in obj);
    document.write('b' in obj);
    document.write('c' in obj);
    document.write('d' in obj);
}

19. 객체 데이터를 펼침연산자(복사)를 통해 불러옵니다.

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    }
    const spread = {...obj}

    document.write(spread.a);
    document.write(spread.b);
    document.write(spread.c);
}

20. 객체 데이터를 펼침연산자(추가)를 통해 불러옵니다.

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    }
    const spread = {...obj, d: "jquery"}

    document.write(spread.a);
    document.write(spread.b);
    document.write(spread.c);
    document.write(spread.d);
}

21. 객체 데이터를 펼침연산자(결합)를 통해 불러옵니다.

{
    const objA = {
        a: 100,
        b: 200
    }
    const objB = {
        c: "javascript",
        d: "jquery"
    }
    const spread = {...objA, ...objB}

    document.write(spread.a);
    document.write(spread.b);
    document.write(spread.c);
    document.write(spread.d);
}

22. 객체 데이터를 객체구조분해할당을 통해 불러옵니다.

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    }
    const { a, b, c } = obj;

    document.write(obj.a);
    document.write(obj.b);
    document.write(obj.c);
}

23. 객체 데이터를 객체구조분해할당을 통해 불러옵니다.

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    }
    const { a:name1, b:name2, c:name3 } = obj;

    document.write(name1);
    document.write(name2);
    document.write(name3);
}
