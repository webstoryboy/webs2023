18. 다음의 결괏값을 작성하시오.

{
    function solution(arr, divisor) {
        var answer = [];
        
        const div = arr.filter( el => el % divisor == 0 );    
        answer = div.length > 0 ? div.sort( (a, b ) => a - b ) : [-1];
        
        return answer;
    }
            
    document.write(solution([1,2,3,4,5], 2))
}
