나에게 코딩이란

졸업 후, 우연한 계기로 접한 코딩은 저에게 새로운 세상을 보여주었습니다. 코딩에는 일정한 답이 있어 무엇을 하면 되는지 확실하게 알 수 있었습니다. 코딩을 배우면서 제가 더 흥미를 가졌던 언어는 자바스크립트였습니다. 더 깊이 알아가고 공부하고 싶어 ‘프론트엔드’ 라는 직군까지 알게되었고, 이에 맞는 역량을 쌓기 위해 끊임없이 노력하는 중입니다.

저는 이런 사람입니다

새로운 것을 배워서 응용하는 것을 좋아하고 그것을 성공할 때의 성취감을 즐깁니다. 안되면 될 때까지 하려는 집요함이 있습니다. 배움에 있어서는 자만하지 않고 늘 부족하다는 생각으로 반복 학습을 하며 기초부터 탄탄히 다지는 것을 중요하게 생각합니다.

앞으로 저는

사용자의 입장에서 생각하여 사용자의 기억에 남을 수 있는 웹 사이트를 만들도록 노력할 것입니다. 또한, 변화가 끊임없이 일어나는 웹 환경에서 최신 트렌드에 뒤처지지 않기 위해 스터디와 같은 모임에 많이 참여하여 꾸준히 공부하며 개인 역량을 키우겠습니다. 이외에도 팀원들과의 원활한 소통으로 서로의 의도를 이해, 존중하여 '함께‘ 최상의 결과물을 만들겠습니다.

function handleTrailer() {
    // 셀렉터 캐시
    var $selector = {
        body: $("body"),
        overlay: $("#blackout"),
        modal: $("#trailerModal"),
        showBtn: $("#showTrailer"),
        hideBtn: $("#hideTrailer")
    };

    // 플레이어
    var player = {
        obj: null, //플레이어 오브젝트
        query: {
            theme: "dark",
            color: "white",
            controls: 1,
            autoplay: 1,
            enablejsapi: 1,
            modestbranding: 0, // 유튜브 로고 감춤
            rel: 0, // 관련 동영상 표시
            showinfo: 0, // 제목, 업로더 감춤
            iv_load_policy: 3, // 특수효과 감춤
        },
        visible: false
    }

    // 보이기, 숨기기 버튼 활성화
    $selector.showBtn.on("click", showPlayer);
    $selector.hideBtn.on("click", hidePlayer);

    // Youtube API를 이용해 iframe생성
    function setPlayer(id) {
        player.obj = new YT.Player('trailer', {
            height: '480',
            width: '282',
            videoId: id,
            playerVars: player.query
        });

        // 처음 플레이어 크기 설정
        resizePlayer();

        // 리사이즈 화면 회전시 플레이어 크기 다시 설정
        $(window).on("resize orientationchange", function () {
            resizePlayer();
        })
    }


    function resizePlayer() {
        var viewport = {},
            frame = {},
            modal = {};

        viewport.width = $(window).width();
        viewport.height = $(window).height();

        frame.width = viewport.width;
        frame.height = frame.width / 1.6; // 16:10 비율의 화면

        modal.top = ((viewport.height - frame.height) / 2) + "px";
        modal.left = "0px";

        $selector.modal.css(modal);
        player.obj.setSize(frame.width, frame.height);
    }

    function showPlayer() {
        if (!player.obj) {
            setPlayer($selector.showBtn.data("yt"));
        }
        $selector.body.addClass("modal_on");
        $selector.overlay.show();
        player.visible = true;
    }

    function hidePlayer() {
        $selector.body.removeClass("modal_on");
        $selector.overlay.hide();
        player.visible = false;
        player.obj.stopVideo();
    }
}

$(function () {
    handleTrailer();
})

        enableClick = false;

    e.preventDefault();

        prevActive();
        enableClick = false;
    }

    e.preventDefault();
    play();

$pauseBtn.on("click", function (e) {
    e.preventDefault();
    stop();

function init() {
    let slideLen = $slide.length
    $slideFrame.css({
        width: 100 * slideLen + "%",
        marginLeft: "-100%"
    });
    $slide.css({
        width: 100 / slideLen + "%"
    })

    $slideFrame.find($slide).last().prependTo($slideFrame);

    play();
}

function nextActive() {
    $slideFrame.stop().animate(
        { marginLeft: "-200%" },
        speed,
        function () {
            $(this).children("div").first().appendTo(this);
            $(this).css({ marginLeft: "-100%" });
            enableClick = true;
        }
    )
}

function prevActive() {
    $slideFrame.stop().animate(
        { marginLeft: "0%" },
        speed,
        function () {
            $(this).children("div").last().prependTo(this);
            $(this).css({ marginLeft: "-100%" });
            enableClick = true;
        }
    )
}

function play() {
    $playBtn.addClass("on");
    $pauseBtn.removeClass("on");

    timer = setInterval(function () {
        nextActive();
    }, 5000);
}

function stop() {
    $pauseBtn.addClass("on");
    $playBtn.removeClass("on")
    clearInterval(timer);
}

function isPwd(name1, name2, len) {
    var pwd1 = $(`[name=${name1}]`).val();
    var pwd2 = $(`[name=${name2}]`).val();
    var regExpPwd = /^(?=.*[a-zA-Z])(?=.*[^a-zA-Z0-9]).{8,16}$/;

    if (pwd1.length >= len && regExpPwd.test(pwd1) && pwd1 == pwd2) {
        $(`[name=${name1}]`).parent().find("p").remove();
        $(`[name=${name2}]`).parent().find("p").remove();
        return true;
    }
    else if (pwd1 !== pwd2) {
        $(`[name=${name2}]`).parent().find("p").remove();
        $(`[name=${name2}]`).parent().append(`
비밀번호가 같지 않습니다 비밀번호를 확인해주세요
`);
        $(`[name=${name2}]`).focus();
        return false;
    }
    else {
        $(`[name=${name1}]`).parent().find("p").remove();
        // $(`[name=${name2}]`).parent().find("p").remove();
        $(`[name=${name1}]`).parent().append(`
입력항목을 양식에 맞춰 ${len}글자 이상 입력하세요
`);
        $(`[name=${name1}]`).focus();
        return false;
    }
}

    let len = $("input.chk").length;
    let chkCnt = 0;

    for (let i = 0; i < len; i++) {
        let chkTF = $(".chk").eq(i).prop("checked");

        if (chkCnt == len) {
